CREATE OR REPLACE Procedure User_Update
(upId int,
upLogin varchar(40),  
upPassword varchar(40),  
upRoleId integer,
upDeptId integer)
language sql
As $$
update Users Set
Login = upLogin,
Password = upPassword,
roleid = upRoleId,
deptid = upDeptId
where Id = upId  
$$ 

       CREATE OR REPLACE Procedure User_Delete
      (uid integer)
       language sql
       AS  $$  
       Delete From users
       where Id = uid
       $$

CREATE OR REPLACE Procedure User_Insert
(inLogin varchar(40),  
 inPassword varchar(40),  
 inRoleId integer,
 inDeptId integer)  
 LANGUAGE SQL
 AS $$  
 Insert into users
(Login, Password, roleid, deptid) Values
 (inLogin, inPassword, inRoleId, inDeptId)
 $$
